{{- /* Local guard for Hextra search indexing. Some generated pages do not expose Fragments, so the upstream partial can fail when it reads .target.Headings. */ -}} {{- with .target -}} {{- range $heading := index .Headings -}} {{- if and (eq $heading.Level 0) (not $heading.Title) -}} {{- $.scratch.Add "keys" (slice $heading.Title) -}} {{- else -}} {{- $key := (printf "%s#%s" $heading.ID $heading.Title) -}} {{- $.scratch.Add "keys" (slice $key) -}} {{- end -}} {{- $title := (printf "%s" $heading.Level $heading.Title) | htmlUnescape -}} {{- $.scratch.Add "titles" (slice $title) -}} {{- partial "utils/extract-headings.html" (dict "target" $heading "scratch" $.scratch ) }} {{- end -}} {{- end -}}
